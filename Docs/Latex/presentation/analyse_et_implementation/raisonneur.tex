\begin{frame}{Raisonneur}{Valuation des formes}
	\begin{itemize}
			\item Valuation d'une forme :
			\[ P(Gain|Forme) = \frac{P(Forme|Gain) \times P(Gain)}{P(Forme)} \]\\[20pt]
			\item Fin de partie $\rightarrow$ mise à jour des formes rencontrées
		\end{itemize}
	
\end{frame}
